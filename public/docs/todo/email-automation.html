1. Email Service Module

File: server/services/email.js

const { Resend } = require('resend');
const fs = require('fs').promises;
const path = require('path');

class EmailService {
  constructor() {
    // Load API key from config
    const config = require('../data/config.json');
    this.resend = new Resend(config.apiKeys.resend);
    this.from = 'Driftless Harvest ';
  }

  async sendOrderConfirmation(order, customer) {
    const html = await this.buildOrderConfirmationEmail(order);
    
    await this.resend.emails.send({
      from: this.from,
      to: customer.email,
      subject: `Order Confirmed - Delivery ${order.deliveryWindow}`,
      html: html
    });
  }

  async sendDeliveryReminder(order, customer) {
    const html = await this.buildDeliveryReminderEmail(order);
    
    await this.resend.emails.send({
      from: this.from,
      to: customer.email,
      subject: `Your delivery arrives tomorrow!`,
      html: html
    });
  }

  async sendPackedNotification(order, customer) {
    const html = await this.buildPackedNotificationEmail(order);
    
    await this.resend.emails.send({
      from: this.from,
      to: customer.email,
      subject: `Your box is packed and ready!`,
      html: html
    });
  }

  buildOrderConfirmationEmail(order) {
    // Build HTML email template
    return `

              \`).join('')}
              

              Total: $${order.total}
            

            
            
Delivery Details

            
When: ${order.deliveryWindow}

            
Where: ${order.address}

            
            
What's Next?

    \`;
  }
}

module.exports = new EmailService();

const cron = require('node-cron');
const emailService = require('../services/email');

// Run daily at 9 AM
cron.schedule('0 9 * * *', async () => {
  console.log('[Cron] Running delivery reminder job');
  
  // Get orders for tomorrow
  const tomorrow = new Date();
  tomorrow.setDate(tomorrow.getDate() + 1);
  
  const orders = await getOrdersForDate(tomorrow);
  
  for (const order of orders) {
    const customer = await getCustomer(order.user_email);
    
    try {
      await emailService.sendDeliveryReminder(order, customer);
      console.log('[Email] Reminder sent to', customer.email);
    } catch (error) {
      console.error('[Email] Failed to send reminder:', error);
    }
  }
});

module.exports = { start: () => console.log('[Cron] Delivery reminders scheduled') };
